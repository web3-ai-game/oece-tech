### Key Pool 管理器範例

```go
// pool.go - Gemini API Key 池管理
package gemini

type KeyPool struct {
    Keys       []string
    UsageMap   map[string]*Usage
    CurrentIdx int
    mu         sync.Mutex
}

type Usage struct {
    RPM        int       // 每分鐘請求數
    RPD        int       // 每天請求數
    LastReset  time.Time
}

func (p *KeyPool) GetNextKey() string {
    p.mu.Lock()
    defer p.mu.Unlock()
    
    // 輪詢策略: 找到使用率最低的 Key
    minUsage := math.MaxInt32
    selectedIdx := 0
    
    for i, key := range p.Keys {
        usage := p.UsageMap[key]
        if usage.RPD < minUsage {
            minUsage = usage.RPD
            selectedIdx = i
        }
    }
    
    p.CurrentIdx = selectedIdx
    return p.Keys[selectedIdx]
}

func (p *KeyPool) RecordUsage(key string) {
    p.mu.Lock()
    defer p.mu.Unlock()
    
    usage := p.UsageMap[key]
    usage.RPM++
    usage.RPD++
    
    // 每分鐘重置 RPM
    if time.Since(usage.LastReset) > time.Minute {
        usage.RPM = 0
        usage.LastReset = time.Now()
    }
}
```
