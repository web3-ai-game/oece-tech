import requests
from bs4 import BeautifulSoup

class HardwarePriceCrawler:
    """東南亞硬件價格爬蟲"""
    
    def __init__(self):
        self.sources = {
            "lazada": "https://www.lazada.sg",
            "shopee": "https://shopee.sg",
            "aliexpress": "https://www.aliexpress.com",
            "amazon": "https://www.amazon.com"
        }
    
    def search_item(self, keyword: str) -> List[Dict]:
        """搜索硬件價格
        
        Args:
            keyword: 搜索關鍵詞 (如 "ESP32 DevKit")
        
        Returns:
            價格列表,按價格排序
        """
        results = []
        
        for source, base_url in self.sources.items():
            try:
                data = self._crawl_source(source, keyword)
                results.extend(data)
            except Exception as e:
                print(f"爬取 {source} 失敗: {e}")
        
        # 按價格排序
        results.sort(key=lambda x: x["price"])
        
        return results
    
    def _crawl_source(self, source: str, keyword: str) -> List[Dict]:
        """爬取單一來源"""
        if source == "lazada":
            return self._crawl_lazada(keyword)
        elif source == "shopee":
            return self._crawl_shopee(keyword)
        elif source == "aliexpress":
            return self._crawl_aliexpress(keyword)
        elif source == "amazon":
            return self._crawl_amazon(keyword)
        return []

        """爬取Lazada"""
        url = f"https://www.lazada.sg/catalog/?q={keyword}"
        headers = {"User-Agent": "Mozilla/5.0"}
        
        response = requests.get(url, headers=headers)
        soup = BeautifulSoup(response.content, "html.parser")
        
        items = []
        for item in soup.select(".gridItem"):
            try:
                title = item.select_one(".title").text.strip()
                price_text = item.select_one(".price").text.strip()
                price = float(re.sub(r"[^\d.]", "", price_text))
                link = item.select_one("a")["href"]
                
                items.append({
                    "title": title,
                    "price": price,
                    "currency": "SGD",
                    "source": "Lazada",
                    "link": link
                })
            except:
                continue
