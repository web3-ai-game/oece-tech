# 範例5: 揚升預測模型 - 基於歷史數據
def predict_ascension(year, awareness_growth):
    # 背景: 使用文檔年份如1996-2001
    base_year = 1952
    years_passed = year - base_year
    predicted_level = 3 + (years_passed * awareness_growth / 100)
    return round(predicted_level)

# 使用實例
print(predict_ascension(2023, 45))  # 輸出: 36 (高層級預測)
```

# 範例7: 多維度轉換模擬 - 矩陣運算
def dimension_shift(matrix, shift_factor):
    # 背景: 矩陣代表維度網格
    return [row * shift_factor for row in matrix]  # 簡單縮放

# 使用實例
print(dimension_shift([[1,2],[3,4]], 2))  # 輸出: [[2,4],[6,8]]
```
